def attendance(events):
    m = 0
    for event in events:
        if event.lower() == 'no':
            return 'N'
        elif event.lower() == 'maybe':
            m = 1
            
    return 'M' if m != 0 else 'Y'

    def x_true(val):
    if val == True:
        return 'X'
    elif val == False:
        return ''
    else:
